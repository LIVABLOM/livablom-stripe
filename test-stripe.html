require('dotenv').config(); const express = require('express'); const cors = require('cors'); const Stripe = require('stripe'); const app = express(); const stripe = Stripe(process.env.STRIPE_SECRET_KEY); // clé Stripe dans .env const PORT = process.env.PORT || 4000; // Détecter si on est en local ou prod const isLocal = process.env.NODE_ENV !== 'production'; const BASE_URL = isLocal ? `http://localhost:${PORT}` : 'https://livablom.fr'; // Middleware app.use(cors()); app.use(express.json()); // Servir les fichiers statiques depuis le dossier public app.use(express.static('public')); // Endpoint pour créer une session Checkout app.post('/create-checkout-session', async (req, res) => { const { date, logement, nuits, prix } = req.body; if (!date || !logement || !nuits || !prix) { return res.status(400).json({ error: 'Paramètres manquants' }); } try { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], line_items: [{ price_data: { currency: 'eur', product_data: { name: `${logement} - ${nuits} nuit(s)`, }, unit_amount: prix * 100, }, quantity: 1, }], mode: 'payment', success_url: `${BASE_URL}/confirmation.html?success=true`, cancel_url: `${BASE_URL}/blom.md`, }); res.json({ url: session.url }); } catch (err) { console.error('Erreur création session Stripe:', err); res.status(500).json({ error: 'Erreur création session Stripe' }); } }); // Démarrage du serveur app.listen(PORT, () => console.log(`Serveur Stripe en écoute sur ${BASE_URL}`));
